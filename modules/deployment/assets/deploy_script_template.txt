#!/bin/bash
# Provision - Deployment Script
# Server    : {{SERVER_NAME}} ({{SERVER_IP}})
# Source    : {{SOURCE_TYPE}}
# Release   : {{RELEASE_PATH}}
# Web root  : {{WEB_ROOT}} (updated on Promote)

set -euo pipefail
umask 022

run_sudo() {
    if [ "$(id -u)" -eq 0 ]; then
        "$@"
        return
    fi
    sudo -n "$@"
}

if [ "$(id -u)" -ne 0 ]; then
    if ! command -v sudo >/dev/null 2>&1; then
        echo "ERROR: sudo is required when deploying as $(whoami)." >&2
        exit 1
    fi
    if ! sudo -n -l >/dev/null 2>&1; then
        echo "ERROR: passwordless sudo is required for $(whoami)." >&2
        exit 1
    fi
fi

# Environment variables from Provision.
{{ENV_VARS}}

SOURCE_TYPE="${SOURCE_TYPE:-{{SOURCE_TYPE}}}"
REPO_URL="${REPO_URL:-{{REPO_URL}}}"
BRANCH="${BRANCH:-{{BRANCH}}}"
WEB_ROOT="${WEB_ROOT:-{{WEB_ROOT}}}"
RELEASE_PATH="{{RELEASE_PATH}}"
DOMAIN="${DOMAIN:-{{DOMAIN}}}"
DB_NAME="${DB_NAME:-{{DB_NAME}}}"

if [ -e "$RELEASE_PATH" ]; then
    echo "ERROR: release path already exists: $RELEASE_PATH" >&2
    exit 1
fi

RELEASE_PARENT=$(dirname "$RELEASE_PATH")
run_sudo mkdir -p "$RELEASE_PARENT"
run_sudo install -d -m 0755 -o "$(id -un)" -g "$(id -gn)" "$RELEASE_PATH"

if [ "$SOURCE_TYPE" = "zip" ]; then
    if [ -z "${DEPLOY_ZIP:-}" ] || [ ! -f "$DEPLOY_ZIP" ]; then
        echo "ERROR: deployment zip is missing." >&2
        exit 1
    fi

    EXTRACT_DIR=$(mktemp -d /tmp/provision_extract_XXXXXX)
    cleanup_zip_deploy() {
        rm -rf "$EXTRACT_DIR"
    }
    trap cleanup_zip_deploy EXIT

    unzip -q "$DEPLOY_ZIP" -d "$EXTRACT_DIR"
    APP_DIR="$EXTRACT_DIR"
    first_child=$(find "$EXTRACT_DIR" -mindepth 1 -maxdepth 1 | head -n 1)
    child_count=$(find "$EXTRACT_DIR" -mindepth 1 -maxdepth 1 | wc -l | tr -d ' ')
    if [ "$child_count" = "1" ] && [ -d "$first_child" ]; then
        APP_DIR="$first_child"
    fi
    cp -a "$APP_DIR"/. "$RELEASE_PATH"/
else
    git clone --branch "$BRANCH" "$REPO_URL" "$RELEASE_PATH"
fi

{{VHOST_BLOCK}}

if [ -n "$DOMAIN" ] && [ -f "$RELEASE_PATH/config/config.php" ]; then
    sed -i "s|define('BASE_URL', '[^']*');|define('BASE_URL', 'https://$DOMAIN/');|" "$RELEASE_PATH/config/config.php"
fi

{{CONFIG_PATCH_BLOCK}}

echo "==> Finalizing release permissions..."
if run_sudo chgrp -R www-data "$RELEASE_PATH"; then
    echo "==> Release group set to www-data"
else
    echo "WARN: could not set release group to www-data; checking release readability..." >&2
    if ! find "$RELEASE_PATH" \( -type f ! -perm -004 -o -type d ! -perm -001 \) -print -quit | grep -q .; then
        echo "WARN: release is world-readable; continuing without group change." >&2
    else
        echo "ERROR: release contains files that Apache may not be able to read." >&2
        exit 1
    fi
fi

DEPLOYED_SHA=$(git -C "$RELEASE_PATH" rev-parse HEAD 2>/dev/null || echo "n/a")
echo "Release staged!"
echo "  Release path: $RELEASE_PATH"
echo "  Web root    : $WEB_ROOT"
echo "  SHA         : $DEPLOYED_SHA"
echo "RELEASE_PATH: $RELEASE_PATH"
echo "SHA: $DEPLOYED_SHA"
